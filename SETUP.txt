=== PRODUCTION DEPLOYMENT ===

gcloud auth login
gcloud config set project virtualcloset-477422

cd virtual-closet-app/server
gcloud run deploy virtual-closet-api \
  --source . \
  --region=us-central1 \
  --allow-unauthenticated

cd ../client
gcloud run deploy virtual-closet-web \
  --source . \
  --region=us-central1 \
  --allow-unauthenticated
